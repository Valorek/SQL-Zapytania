1.SELECT kategorie.OpisKategorii, COUNT(produkty.NumerProduktu) FROM kategorie JOIN produkty ON kategorie.IDKategorii=produkty.IDKategorii GROUP BY kategorie.OpisKategorii; 
2.SELECT dostawcy.NazwaDostawcy, COUNT(dostawcy_produktow.NumerProduktu) FROM dostawcy JOIN dostawcy_produktow ON dostawcy.IDDostawcy=dostawcy_produktow.IDDostawcy GROUP BY dostawcy.NazwaDostawcy; 
3.SELECT klienci.ImieKlienta, klienci.NazwiskoKlienta, COUNT(zamowienia.NumerZamowienia) FROM klienci JOIN zamowienia ON klienci.IDKlienta=zamowienia.IDKlienta GROUP BY klienci.ImieKlienta; 
4.SELECT klienci.ImieKlienta, klienci.NazwiskoKlienta, COUNT(zamowienia.NumerZamowienia) FROM klienci JOIN zamowienia ON klienci.IDKlienta=zamowienia.IDKlienta GROUP BY klienci.ImieKlienta HAVING COUNT(zamowienia.NumerZamowienia)<30; 
5.SELECT klienci.ImieKlienta, klienci.NazwiskoKlienta, COUNT(zamowienia.NumerZamowienia) FROM klienci JOIN zamowienia ON klienci.IDKlienta=zamowienia.IDKlienta WHERE zamowienia.DataZamowienia BETWEEN '2012-12-01' AND '2012-12-31' GROUP BY klienci.ImieKlienta;
6.SELECT klienci.ImieKlienta, klienci.NazwiskoKlienta, SUM(dane_zamowien.CenaZakupu) FROM klienci JOIN zamowienia ON zamowienia.IDKlienta=klienci.IDKlienta JOIN dane_zamowien ON zamowienia.NumerZamowienia=dane_zamowien.NumerZamowienia GROUP BY klienci.ImieKlienta; 
7.SELECT klienci.ImieKlienta, klienci.NazwiskoKlienta, SUM(dane_zamowien.CenaZakupu) FROM klienci JOIN zamowienia ON zamowienia.IDKlienta=klienci.IDKlienta JOIN dane_zamowien ON zamowienia.NumerZamowienia=dane_zamowien.NumerZamowienia WHERE MONTH(zamowienia.DataZamowienia)=12 AND YEAR(zamowienia.DataZamowienia)=2012 GROUP BY klienci.ImieKlienta ORDER BY SUM(dane_zamowien.CenaZakupu) DESC; 
8.SELECT klienci.ImieKlienta, klienci.NazwiskoKlienta, SUM(dane_zamowien.CenaZakupu*dane_zamowien.ZamowionaIlosc) FROM klienci JOIN zamowienia ON zamowienia.IDKlienta=klienci.IDKlienta JOIN dane_zamowien ON zamowienia.NumerZamowienia=dane_zamowien.NumerZamowienia GROUP BY klienci.ImieKlienta DESC HAVING SUM(dane_zamowien.CenaZakupu*dane_zamowien.ZamowionaIlosc)>200000; 
9.SELECT produkty.NazwaProduktu, SUM(dane_zamowien.CenaZakupu*dane_zamowien.ZamowionaIlosc) FROM produkty JOIN dane_zamowien ON produkty.NumerProduktu=dane_zamowien.NumerProduktu JOIN zamowienia ON dane_zamowien.NumerZamowienia=zamowienia.NumerZamowienia JOIN klienci ON zamowienia.IDKlienta=klienci.IDKlienta GROUP BY produkty.NazwaProduktu;
